# Escanear instalados na categoria libs e gerar plano (sem aplicar), com JSON:
/usr/src/adm/scripts/10.20-bulk-update-planner.sh --scope installed --categories libs --json

# Rodar em metafiles (cat/apps + cat/libs), permitir pré-releases, baixar fontes e calcular SHA-256, salvar metafiles e ordenar por grafo:
/usr/src/adm/scripts/10.20-bulk-update-planner.sh \
  --scope metafiles --categories "apps,libs" \
  --include-prerelease --download --apply --parallel 6 --dl-parallel 8

# A partir de uma lista (/tmp/alvo.lst com linhas cat/name), excluir alguns padrões e não filtrar “apenas novos”:
/usr/src/adm/scripts/10.20-bulk-update-planner.sh \
  --scope list --list-file /tmp/alvo.lst --exclude "apps/*-beta,libs/zz-*" --all

# Respeitar world.lst apenas na ordem (não impede atualização, só prioriza):
/usr/src/adm/scripts/10.20-bulk-update-planner.sh --respect-world

# (opcional) escolher libc
export PROFILE_LIBC=glibc   # ou: musl

# 1) Linux API headers
adm install sys/linux-headers

# 2) binutils (passo 1)
adm install sys/binutils-pass1

# 3) libs pro GCC
adm install libs/gmp
adm install libs/mpfr
adm install libs/mpc
adm install libs/isl
adm install libs/zlib

# 4) gcc (passo 1, sem libc)
adm install sys/gcc-pass1 --with-build

# 5) libc headers/startfiles (glibc|musl)
adm install sys/libc-headers

# 6) binutils (passo 2)
adm install sys/binutils-pass2

# 7) gcc (passo 2, final)
adm install sys/gcc-pass2 --with-build

# Kernel padrão com config da distro (se você copiar pra config.base):
adm install sys/linux-kernel

# Kernel agressivo (Clang/LLVM, LTO):
# (opcional) crie config.base e/ou config.aggressive ao lado do metafile para ajustar features
adm install sys/linux-kernel-aggressive

# Forçar sufixo de versão (LOCALVERSION):
LOCALVERSION="-custom1" adm install sys/linux-kernel
LOCALVERSION="-aggr1"  adm install sys/linux-kernel-aggressive

# Instalar tudo (full)
adm install sys/linux-firmware

# Instalar só alguns firmwares (minimal)
# Exemplo: só placas Intel (iwlwifi) + AMDGPU:
export FIRMWARE_MODE=minimal
export FIRMWARE_INCLUDE="iwlwifi amdgpu radeon"
adm install sys/linux-firmware

# Excluir coisas muito grandes (ex.: DSP de som que você não usa):
export FIRMWARE_MODE=minimal
export FIRMWARE_INCLUDE="iwlwifi amdgpu"
export FIRMWARE_EXCLUDE="brcm nvidia"
adm install sys/linux-firmware
